Bootstrap Categories Demo

Basic

$('#demo-0').categories(data);

Addable and removable

$('#demo-1').categories(data, {
    addable: true,
    removable: true
});

Events

$('#demo-2').categories(data, {
    addable: true,
    removable: true,

    onAdd: function(category) {
        alert('You added ' + category.name);
    },
    onRemove: function(category) {
        alert('You removed ' + category.name);
    },
    onSelect: function(category) {
        alert('You selected ' + category.name);
    }
});

Custom styling

$('#demo-3').categories(data, {
    addable: true,
    removable: true,

    activeClass: 'list-group-item-success',
    addButtonClass: 'btn btn-primary',
    addButtonHtml: 'Add',
    removeButtonClass: 'btn btn-xs btn-warning pull-right',
    removeButtonHtml: 'Remove',
    addInputPlaceholder: 'New category'
});
